class SmartDevice {
protected:                        // التغليف: محمية من الخارج
    string deviceName;
    int    powerConsumption;       // بالواط
    bool   isOn;

public:
    // الباني: يهيئ الجهاز عند إنشائه
    SmartDevice(string name = "Unknown", int power = 0) {
        deviceName        = name;
        powerConsumption  = power;
        isOn              = false;   // الافتراضي: مطفي
    }

    void turnOn()  { isOn = true;  }
    void turnOff() { isOn = false; }

    void showStatus() {
        cout << "- " << deviceName
             << " is " << (isOn ? "ON" : "OFF") << endl;
    }
};

التحميل الزائد — Operator Overloading

operator+.cpp C++ 
// داخل صف SmartDevice
int operator+(SmartDevice d) {

    // ✅ الإصلاح المنطقي:
    // يحسب فقط الأجهزة الشغّالة
    int p1 = isOn ? powerConsumption : 0;
    int p2 = d.isOn ? d.powerConsumption : 0;

    return p1 + p2;
}

// الاستخدام في main:
int total = ac + light;
// بدلاً من:
// ac.powerConsumption + light.powerConsumption

main.cpp — Smart Home Load Management C++ 
#include <iostream>
#include <string>
using namespace std;

// ══════════════════════════════════════
//  الصف الأساسي — SmartDevice
// ══════════════════════════════════════
class SmartDevice {
protected:
    string deviceName;
    int    powerConsumption;
    bool   isOn;

public:
    SmartDevice(string name = "Unknown", int power = 0)
        : deviceName(name), powerConsumption(power), isOn(false) {}

    void turnOn()  { isOn = true;  }
    void turnOff() { isOn = false; }

    void showStatus() {
        cout << "- " << deviceName
             << " is " << (isOn ? "ON" : "OFF") << endl;
    }

    // ✅ التحميل الزائد — يحسب الأجهزة الشغّالة فقط
    int operator+(SmartDevice d) {
        int p1 = isOn ? powerConsumption : 0;
        int p2 = d.isOn ? d.powerConsumption : 0;
        return p1 + p2;
    }
};

// ══════════════════════════════════════
//  الوراثة — AirConditioner
// ══════════════════════════════════════
class AirConditioner : public SmartDevice {
private:
    int temperature;

public:
    AirConditioner(string name, int power, int temp)
        : SmartDevice(name, power), temperature(temp) {}

    void setTemperature(int temp) {
        temperature = temp;
        cout << deviceName << " temp set to: "
             << temperature << "°C" << endl;
    }
};

// ══════════════════════════════════════
//  الوراثة — SmartLight
// ══════════════════════════════════════
class SmartLight : public SmartDevice {
private:
    int brightness;

public:
    SmartLight(string name, int power)
        : SmartDevice(name, power), brightness(100) {}

    void setBrightness(int level) {
        brightness = level;
        cout << deviceName << " brightness: "
             << brightness << "%" << endl;
    }
};

// ══════════════════════════════════════
//  نقطة التشغيل
// ══════════════════════════════════════
const int MAX_LOAD = 2000;    // ✅ ثابت بدلاً من رقم مكتوب مباشرة

int main() {
    AirConditioner ac   ("Living Room AC", 2000, 24);
    SmartLight     light ("Bedroom Light",  50);

    ac.turnOn();
    ac.setTemperature(22);

    light.turnOn();
    light.setBrightness(75);

    int totalPower = ac + light;
    cout << "\nTotal Power Consumption: "
         << totalPower << " Watts." << endl;

    if (totalPower > MAX_LOAD) {
        cout << "[WARNING]: High power load detected!" << endl;
    }

    return 0;
}
